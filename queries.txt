// DEVUELVE LA MAC, EDAD, SEXO DE LOS USUARIOS QUE TIENEN TELEFONO

SELECT usuario.mac AS MAC, rasgos.edad AS Edad, rasgos.sexo AS Sexo
FROM usuario
INNER JOIN usuariorasgo ON usuario.mac = usuariorasgo.usuariofk
INNER JOIN rasgos ON usuariorasgo.rasgosfk = rasgos.idrasgos

// DEVUELVE LA CANTIDAD DE USUARIOS DE CADA  SEXO HAN ENTRADO AL CENTRO COMERCIAL
SELECT COUNT(rasgos.idrasgos) as Cantidad, rasgos.sexo AS Sexo
FROM rasgos
GROUP BY rasgos.sexo
ORDER BY Cantidad DESC
